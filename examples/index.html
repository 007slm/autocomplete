seajs.use(['../src/autocomplete', '$'], function(AutoComplete, $) {
    var data = [
        '163.com',
        '126.com',
        'gmail.com'
    ];
    new AutoComplete({
        trigger: '#example',
        dataSource: function(query) {
            return $.map(data, function(v, i) {
                return query + '@' + v;
            });
        },
        filter: '',
        inputFilter: function(v){
            return v.replace(/^(.*)@.*$/,'$1');
        }
    }).render();
});
